ACCION 1.1.17 (ProductoSumado) ES
AMBIENTE
A,B,,i,Result: ENTERO
Bandera:Alfanumerico

FUNCION SUMADOS(X,Y:ENTERO):ENTERO ES
	SI Y>0 ENTONCES 
		PARA i:=1 hasta Y HACER 
			SUMADOS:=SUMADOS+X
		FIN PARA
	CONTRARIO
		PARA i:=1 hasta -Y HACER 
			SUMADOS:=SUMADOS+(-X) 
		FIN PARA
	FIN SI
FIN FUNCION

PROCESO
	Mientras Bandera='Y' HACER
		Result:=0
		ESCRIBIR("INGRESE LOS DOS ENTEROS A MULTIPLICAR")
		LEER(A,B) -5 -6
		Result:=SUMADOS(A,B)
		ESCRIBIR("EL RESULTADO DE LA MULTIPLICACION ES, Result)
		ESCRIBIR("Repetir programa? [Y/N]")
		LEER(Bandera)
	FIN MIENTRAS
FIN ACCION
